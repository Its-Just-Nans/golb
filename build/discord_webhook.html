Discord Webhook

Using Browser

const url = "https://discord.com/api/webhooks/..."
const data = {
    content: "my content",
    username: "log",
    avatar_url: "",
    embeds: [] // for many small messages
}
fetch(url, {
    method : "POST"
    body: data
});

Using NodeJS

const axios = require("axios");

const url = "https://discord.com/api/webhooks/..."
const data = {
    content: "my content",
    username: "log",
    avatar_url: "",
    embeds: [] // for many small message
}
axios.post(url, data);
